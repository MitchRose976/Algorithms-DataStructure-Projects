function checkCashRegister(price, cash, cid) {
  
  //sorts cid in order from highest currency value to lowest currency value
  let cidHighestToLowest = cid.reverse(); 
  //matches each currency type in cidHighestToLowest with appropriate amount
  let currencyValues = [100,20,10,5,1,0.25,0.1,0.05,0.01];  
  let changeDue = cash - price;
  //seperate array for amount of change to give
  let changeOwing = [];
  //format to return 
  let result = {status:"", change: []};
  //calculates total amount of cash in cid
  let totalCid = cid.flat().filter(item => typeof item == "number").reduce(  (accumulator, currentValue) => {
      return accumulator + currentValue;
    }, 0)

  console.log()

  if (totalCid > changeDue) {
    for (let i = 0; i < currencyValues.length; i++) {
      while (changeDue > currencyValues[i]) {
        /*if total amount of currency type in cid is greater than changeDue,
          push max amount of that currency that is less than changeDue*/
        if (cidHighestToLowest[i][1] > changeDue) { 
            changeOwing.push([cidHighestToLowest[i][0], Math.floor(changeDue / currencyValues[i]) * currencyValues[i]]);        
              changeDue = changeDue - Math.floor(changeDue / currencyValues[i]) * currencyValues[i]; 
              changeDue = Math.round(changeDue * 100) / 100; 
              i++
        /*if changeDue is greater than total amount of currency type in cid, 
          push entire value to changeOwing and subtract from changeDue*/
        } else if (changeDue > cidHighestToLowest[i][1]) {
              changeOwing.push(cidHighestToLowest[i]);
              changeDue = changeDue - cidHighestToLowest[i][1];  
              i++
        //if exact change is returned     
        } if (changeDue === 0) {
          result.change = changeOwing;
          result.status = "OPEN";
        //if cannot return exact change 
        } else {
          result.status = "INSUFFICIENT_FUNDS";
          result.change = [];
        }
      }
    } 
  } else if (totalCid === changeDue) {
      changeDue = 0;
      result.change = cid.reverse();
      result.status = "CLOSED";
  } else {
    result.status = "INSUFFICIENT_FUNDS";
  }
  return result;
}
